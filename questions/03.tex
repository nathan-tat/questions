% Question 3

\begin{question}
    Prove \[ p^2 + q^2 \leq (p+q)^2 \] for \(0 \leq q \) and \(0 \leq p \). \pts{4}
\end{question}

\begin{solution}
    We are going to show \(0 \leq \text{RHS} - \text{LHS}\).

    \begin{align*}
        \text{RHS} - \text{LHS} &= (p+q)^2 - p^2 - q^2 \\
        &= p^2 + 2pq + q^2 - p^2 - q^2 \\
        &= 2pq \\
        &\geq 0 ~~ \because p \geq 0, ~ q \geq 0 \qed
    \end{align*}
\end{solution}
